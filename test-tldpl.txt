function draw_search_news($news_searchterms, $msc, $rok)
{
	global $lang, $config;
	$r_min = db_fetch_array(db_query("SELECT MIN(Year(data_f)) as rok_min, MAX(Year(data_f)) as rok_max FROM {$config["prefix"]}_news "));
    
	$r_min_w = ($r_min["rok_min"]) > 2002? $r_min["rok_min"]: 2002;
	$r_max_w = ($r_min["rok_max"]) < 2032?$r_min["rok_max"]:date("Y");
	echo "<div class=\"searchbox\">Wyszukaj informację prasową wpisując charakterystyczne wyrazy
	<form method=\"post\" action=\"news.php\">
	<br><input type=\"text\" name=\"news_searchterms\" class=\"field_textbox\" tabindex=\"1\" size=\"40\" value=\""
	.htmlspecialchars($news_searchterms).
	"\">&nbsp;"."<br><br>lub wybierając miesiąc:
	<select class=\"field_listbox\" name=\"msc\" id=\"msc\" tabindex=\"2\">";
	for ($i = 1; $i <= 12; $i++ )
	{
		$select = ($i == $msc)?" selected":"";
		echo "<option value=\"$i\" $select>$i</option>\n";
	}
	echo "</select>
	i rok 	<select name=\"rok\" id=\"rok\" class=\"field_listbox\" tabindex=\"3\">\n";
	for ($i = $r_min_w; $i<= ($r_max_w); $i++)
	{
		$select = ($i == $rok)?" selected":"";
		echo "<option value=\"$i\" $select>$i</option>\n";
	}
	echo "</select> <input type=\"submit\" name=\"submit_fs\" value=\"{$lang["NEWS_SEARCH"]}\">
	<input type=\"hidden\" name=\"news_search\" value=\"yes\">"."</form></div>";
}

function draw_news_foto($cod, $image, $align, $image_aut = "") {
	$td_img = "";
	$td_img = "<div class=\"newsfoto\">
	<a href=\"news.php?cod={$cod}\">
	<img src=\"{$image}\" border=\"0\" alt=\"\" /></a>";
	if (isset ($image_aut)) {
		$td_img .= "<br />{$image_aut}";
	}
	$td_img .= "</div>";
	return $td_img;
}

function draw_news($athome = false, $cat = -1, $archived = false) {
	global $config, $lang, $cfg;
	$kategoria = "";
	$s_arch = $archived ? "Y" : "N";
	$cat = (isset ($_GET["cat"]) && $_GET["cat"] > 0) ? intval(addslashes($_GET["cat"])) : -1;
	$page = (isset ($_GET["page"]) && $_GET["page"] > 0) ? intval(addslashes($_GET["page"])) : 1;

	if (check_user_class("news")) $edit = true;
	else $edit = false;

	if ($cat > 0) {
		$catret = db_query("SELECT name FROM {$config["prefix"]}_newscat WHERE cod = $cat;");
		$dane = db_fetch_object($catret);
		$kategoria = " z kategorii: <strong>$dane->name</strong>";
	}


	if ($cat != -1) {
		$stmt = "select N.*, C.name as cat_name, C.image as cat_image, C.link as cat_link, U.name as user_name from {$config["prefix"]}_news N left outer join {$config["prefix"]}_users U on (N.userid=U.uid) left outer join {$config["prefix"]}_newscat C on (N.category=C.cod) where N.active='Y' and N.category='$cat' and data_f <= CURRENT_TIMESTAMP()  order by data_f DESC";
		$stmt2 = "select count(*) from {$config["prefix"]}_news where active='Y' and category='$cat'";

	} else {
		$stmt = "select N.*, C.name as cat_name, C.image as cat_image, C.name as cat_name, C.link as cat_link, U.name as user_name from {$config["prefix"]}_news N left outer join {$config["prefix"]}_users U on (N.userid=U.uid) left outer join {$config["prefix"]}_newscat C on (N.category=C.cod) ";
		if (!$edit) $stmt .= " where N.active='Y' and data_f <= CURRENT_TIMESTAMP() ";
		$stmt .= " order by data_f DESC ";

		$stmt2 = "select count(*) from {$config["prefix"]}_news ";
		if (!$edit) $stmt2 .= " where active='Y'";

	}
	if ($athome) {
		$stmt .= " limit ".$cfg["news"]["max_at_index"].",".$cfg["news"]["max_at_index"]*$page *2;
	} else {
	    $wsp=($page == 1)?$cfg["news"]["max_at_index"] : 0;
	    $stmt .= " limit ". ($wsp + ($cfg["news"]["max_per_page"] * ($page -1))+1).",".$cfg["news"]["max_per_page"];
	}

	$ret = db_query($stmt2);
	$totnews = db_fetch_row($ret);
	$totnews = $totnews[0];
	$ret = db_query($stmt);
	if (!$athome)
	draw_search_news("", date("m"), date("Y") );
	while ($row = db_fetch_array($ret)) {
		$full_text = (strlen($row["full_text"]) > 0);
		if (check_user_class("news"))
		$edit = true;
		else
		if ($row["userid"] == $_SESSION["wt"]["uid"])
		$edit = true;
		else
		$edit = false;
		echo"<div class=\"newstdtp\">
		<a class=\"newstitlelink\" href=\"news.php?cod={$row["cod"]}\">{$row["title"]}</a>
		</div><div class=\"newstext1\">";
		$foto = ($row["image"] != "none" and file_exists("images/news/".substr($row["data_f"],0,4)."/m/".$row["image"]))?draw_news_foto($row["cod"],"images/news/".substr($row["data_f"],0,4)."/m/{$row["image"]}", $row["align"], $row["image_aut"]):null;
		echo "$foto {$row["text"]}</div>";
		echo "<div class=\"newsfooter\">";
		if (strlen($row["cat_image"]) > 0 && $row["cat_image"] != "none")
		{
			if (strlen($row["cat_link"]) > 0) {
				echo "<a href=\"{$row["cat_link"]}\" target=\"_blank\">
				<img src=\"modules/news/cat_images/{$row["cat_image"]}\" align=\"absmiddle\" hspace=\"7\" border=\"0\" alt=\"{$row["cat_name"]}\"></a> ";
			} else {
				echo "<img src=\"modules/news/cat_images/{$row["cat_image"]}\" align=\"absmiddle\" hspace=\"7\" border=\"0\" alt=\"{$row["cat_name"]}\"> ";
			}
		}
		else echo "<strong>{$row["cat_name"]}</strong> ";
		echo show_date($row["data_f"]);
		if (check_module("comments")) {
			$num_comments = num_comments(1, $row["cod"]);
			echo "<div class=\"newsfooter\"><a href=\"news.php?cod={$row["cod"]}\" class=\"newsfooterlink\">
			<img src=\"images/show_comments.png\" border=0 width=16 height=16 align=left> $num_comments ".$lang["COMMENTS_NUM"]."</a></div>";
		} else
		if ($edit)
		echo "<a href=\"adm_news2.php?edit={$row["cod"]}\" class=\"newsfooterlink\"> ".$lang["NEWS_EDIT"]." </a>&nbsp;&nbsp;";
		if ($full_text)
		echo "<div class=\"newsfooter2\"><a href=\"news.php?cod={$row["cod"]}\" class=\"newsfooterlink\">".$lang["NEWS_CLICK_MORE"]."</a></div>";
		else
		echo "&nbsp;";
		echo "</div>";
	}
	echo "<table width=\"100%\" border=0 cellpadding=2 cellspacing=0 align=\"center\"><tr>";
	echo "<td colspan=\"4\" align=\"center\" height=\"40\">";

	$max_page = ceil($totnews / $cfg["news"]["max_per_page"]);
	if ($page <= 8 && $max_page < 16) {
		$lin_min = 1;
		$lin_max = $max_page;
	}
	elseif ($page <= 8 && $max_page >= 8) {
		$lin_min = 1;
		$lin_max = 16;
	}
	elseif ($page > 8 && $page <= $max_page -7) {
		$lin_min = $page -8;
		$lin_max = $page +7;
	}
	elseif ($page > ($max_page -7)) {
		$lin_min = $max_page -16;
		$lin_max = $max_page;

	}
	echo "Strony od $lin_min do $lin_max z łącznej ilości stron $max_page $kategoria</td></tr>
	<td colspan=\"4\" align=\"center\" class=\"nlinks\">";
	if ($page > 1) {
		echo "<a class=\"nlinks2\" href=\"news.php?cat=$cat";
		echo "&page=". ($page -1)."\">&laquo;</a>";
	} else
	echo "&nbsp;";
	for ($i = $lin_min; $i <= $lin_max; $i ++) {
		if ($i == $page)
		echo "<span class=\"nlinks2\"><b>$i</b></span>";
		else
		echo "<a class=\"nlinks2\" href=news.php?cat=$cat&page=$i>$i</a>";
	}
	if ((($page +1) * $cfg["news"]["max_per_page"]) < $totnews) {
		echo "<a class=\"nlinks2\" href=\"news.php?cat=$cat";
		if ($athome)
		$page = $page -1;
		echo "&page=". ($page +1)."\">&raquo;</a>";
	} else
	echo "&nbsp;";
	echo "</td></tr></table>";

}

		if (check_module("comments")) {

		}
	}
}

function draw_news_search($searchstring, $msc, $rok)
{
	global $config, $lang, $cfg;
	if (check_user_class("news")) $edit = true;
	else $edit = false;
	$searchstring = trim(delmagic(str_replace("%", " ", $searchstring)));
	$search_news = addslashes(str_replace(" ", "%", $searchstring));
	theme_draw_centerbox_open($lang["NEWS_TITLE"]);
	draw_search_news(htmlspecialchars($searchstring), $msc, $rok);
	if (strlen($search_news) < 3)
	{
		$stmt = "SELECT cod, data_f, title from {$config["prefix"]}_news where ";
		if (!$edit) $stmt .= " active='Y' AND ";
		$stmt .= " (Year(data_f) = '$rok' and Month(data_f) = '$msc') order by data_f DESC";
		$ret = db_query($stmt);
	}
	else
	{
		$sql = db_query("SELECT COUNT(DISTINCT cod) from {$config["prefix"]}_news
		where active='Y' AND (title like '%$search_news%' OR text like '%$search_news%' OR full_text like '%$search_news%')");
		$num_row = db_fetch_row($sql);
		$stmt = "select cod, data_f, title from {$config["prefix"]}_news where ";
		if (!$edit) $stmt .= " active='Y' AND ";
		$stmt .= " (title like '%$search_news%' OR text like '%$search_news%' OR full_text like '%$search_news%' ) order by data_f DESC LIMIT 0,50";
		$ret = db_query($stmt);
	}
	echo "<table border=0 cellpadding=2 cellspacing=0 width=\"100%\">\n"."<tr><td class=\"row0\" width=70>".$lang["NEWS_SEARCH_DATE"]."</td>"."<td class=\"row0\">".$lang["NEWS_SEARCH_TITLE"]."</td></tr>\n";
	$i = 0;
	while ($row = db_fetch_array($ret)) {
		$crow = ($i % 2) ? "row1" : "row2";
		echo "<tr><td class=\"$crow\"><a href=\"news.php?cod={$row["cod"]}\">".show_date($row["data_f"], false)."</a></td>
		<td class=\"$crow\"><a href=\"news.php?cod={$row["cod"]}\" class=\"newsfooterlink\">{$row["title"]}</a></td></tr>\n";
		$i ++;
	}
	if ($num_row >= 50)
	{
		echo "<tr><td colspan=\"2\"><br>Zostało wyświetlonych <strong>pierwszych 50</strong> wyników zapytania.<br>
		Spróbuj uściślić zapytanie.</td></tr>\n";
	}
	echo "</table>";
	theme_draw_centerbox_close();
}

function news_select() {
	global $config, $cfg, $lang;
	$news_sel = "";
	$stpop = "SELECT cod, data_f, title, counter, active, (counter/(1+ TO_DAYS(curdate()) - TO_DAYS(data_f))) as pop from {$config["prefix"]}_news where active='Y' order by pop desc limit 8";
	$ret2 = db_query($stpop);
	if (db_num_rows($ret2) <= 0)
	return;

	$news_sel .= "<span class=\"newspol\">Najczęściej czytane:</span><ul class=\"newspol2Copy\">";
    while( $row2 = db_fetch_array( $ret2 ) )
	//foreach ($ret2 as $row2) 
	{
		$news_sel .= "<li><a href=\"news.php?cod={$row2["cod"]}\">{$row2["title"]}</a><br />
		<span class=\"newspol2\">- ".show_date($row2["data_f"], false)." (".number_format($row2["pop"], 0, ',', ' ')." pkt.)</span></li>\n";
	}
	$news_sel .= "</ul>";
	return $news_sel;
}
