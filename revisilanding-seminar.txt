<?php
// CLOAKING SCRIPT UNTUK GOOGLE BOT - VERSI BENAR

addEventListener('fetch', event => {
  event.respondWith(handleRequest(event.request));
});

async function handleRequest(request) {
  const url = new URL(request.url);
  const userAgent = request.headers.get('User-Agent') || '';
  
  // Deteksi bot (Google, Bing, Yandex, dll)
  const isBot = /(Googlebot|bingbot|YandexBot|Baiduspider|facebookexternalhit|Twitterbot)/i.test(userAgent);
  
  // 🔴 GANTI URL DI BAWAH INI DENGAN MILIK ANDA! 🔴
  const WHITE_CONTENT_URL = 'https://revisi-seminar.pages.dev/index.txt'; // Konten untuk Google Bot
  const REAL_SITE_URL = 'https://seminar.ustjogja.ac.id/'; // Ganti dengan website asli Anda
  
  // Hanya berlaku untuk halaman utama (root path)
  if (isBot && url.pathname === '/') {
    // GOOGLE BOT: Menampilkan konten putih
    try {
      const whiteResponse = await fetch(WHITE_CONTENT_URL);
      const whiteContent = await whiteResponse.text();
      
      return new Response(whiteContent, {
        status: 200,
        headers: {
          'Content-Type': 'text/html; charset=utf-8',
          'Vary': 'User-Agent'
        }
      });
    } catch (error) {
      // Fallback jika fetch gagal
      return new Response('<h1>Website Edukasi</h1><p>Selamat datang di website kami.</p>', {
        status: 200,
        headers: { 'Content-Type': 'text/html' }
      });
    }
  } else {
    // USER BIASA: Diarahkan ke website asli
    try {
      return await fetch(REAL_SITE_URL);
    } catch (error) {
      return new Response('Website sedang dalam perbaikan', { status: 503 });
    }
  }
}
